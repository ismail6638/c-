using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace ConsoleAph_h3
{
    class Program
    {
        static void Main(string[] args)
        {
            int toplam = 3;

            Person[] kisiler = new Person[toplam];

            for (int i = 0; i < toplam; i++)
            {
                Console.Write("Kişi Adı Giriniz :");
                kisiler[i] = new Person();
                kisiler[i].Name = Console.ReadLine();
            }

            for (int i = 0; i < toplam; i++)
            {
                Console.WriteLine(kisiler[i].ToString());
            }

            Console.ReadKey();
        }
    }

    class Person
    {
        public string Name;

        public override string ToString()
        {
            return "Merhaba benim adım " + Name;
        }
    }
}
